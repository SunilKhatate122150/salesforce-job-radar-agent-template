Sibling components on the same record page cannot use CustomEvent (that is only parent-child). Use Lightning Message Service (LMS).

        <fieldName>loanId</fieldName><description>Loan record Id</description>

    </lightningMessageFields>
</LightningMessageChannel>
// STEP 2: Component A — PUBLISHER (e.g., Status Update button component)
import { LightningElement, api, wire } from 'lwc';
import { publish, MessageContext } from 'lightning/messageService';
import LOAN_UPDATE_CHANNEL from '@salesforce/messageChannel/LoanUpdate__c';
import updateStatus from '@salesforce/apex/LoanController.updateStatus';
import { ShowToastEvent } from 'lightning/platformShowToastEvent';

export default class LoanStatusUpdater extends LightningElement {
    @api recordId;
    @wire(MessageContext) messageContext;
    newStatus = '';

    handleStatusChange(e) { this.newStatus = e.detail.value; }

    async handleSave() {
        try {
            await updateStatus({ loanId: this.recordId, status: this.newStatus });
            // Publish — all subscribers on this page will receive this
            publish(this.messageContext, LOAN_UPDATE_CHANNEL, {
                loanId: this.recordId,
                status: this.newStatus
            });
            this.dispatchEvent(new ShowToastEvent({ title: 'Updated', variant: 'success' }));
        } catch(e) {
            this.dispatchEvent(new ShowToastEvent({ title: 'Failed', message: e.body?.message, variant: 'error' }));
        }
    }
}
// STEP 3: Component B — SUBSCRIBER (e.g., Loan Timeline component)
import { LightningElement, api, wire } from 'lwc';
import { subscribe, unsubscribe, APPLICATION_SCOPE, MessageContext } from 'lightning/messageService';
import { refreshApex } from '@salesforce/apex';
import LOAN_UPDATE_CHANNEL from '@salesforce/messageChannel/LoanUpdate__c';
import getLoanTimeline from '@salesforce/apex/LoanController.getLoanTimeline';

export default class LoanTimeline extends LightningElement {
    @api recordId;
    @wire(MessageContext) messageContext;
    subscription = null;
    wiredTimelineResult;

    @wire(getLoanTimeline, { loanId: '$recordId' })
    wiredTimeline(result) { this.wiredTimelineResult = result; }

    connectedCallback() {
        this.subscription = subscribe(
            this.messageContext, LOAN_UPDATE_CHANNEL,
            (msg) => this.handleMessage(msg),
            { scope: APPLICATION_SCOPE }
        );
    }

    async handleMessage(msg) {
        if(msg.loanId !== this.recordId) return;
        // Status updated — refresh timeline data
        await refreshApex(this.wiredTimelineResult);
    }

    disconnectedCallback() {
        unsubscribe(this.subscription); // ALWAYS unsubscribe — prevent memory leaks
    }
}
Complete getRecord + getFieldValue implementation — all patterns explainedRECORD PAGE
// loanDetailsWidget.js
import { LightningElement, api, wire } from 'lwc';
import { getRecord, getFieldValue } from 'lightning/uiRecordApi';

// ALWAYS import fields using schema imports — never hardcode 'Loan__c.Status__c'
import LOAN_STATUS  from '@salesforce/schema/Loan__c.Status__c';
import LOAN_AMOUNT  from '@salesforce/schema/Loan__c.LoanAmount__c';
import LOAN_TYPE    from '@salesforce/schema/Loan__c.LoanType__c';
import OFFICER_NAME from '@salesforce/schema/Loan__c.LoanOfficer__r.Name'; // related field
import OFFICER_EMAIL from '@salesforce/schema/Loan__c.LoanOfficer__r.Email';

const FIELDS = [LOAN_STATUS, LOAN_AMOUNT, LOAN_TYPE, OFFICER_NAME, OFFICER_EMAIL];

export default class LoanDetailsWidget extends LightningElement {
    @api recordId;
    error;
    wiredRecord;

    // Pattern 1: Wire as function (gives you both data and error)
    @wire(getRecord, { recordId: '$recordId', fields: FIELDS })
    wiredLoan(result) {
        this.wiredRecord = result;  // store for refreshApex
        if(result.error) this.error = result.error.body?.message;
    }

    // Pattern 2: Wire as property (simpler syntax)
    @wire(getRecord, { recordId: '$recordId', fields: FIELDS })
    loan; // this.loan.data and this.loan.error

    // Use getFieldValue to safely extract values — handles null/undefined gracefully
    get loanStatus()  { return getFieldValue(this.loan.data, LOAN_STATUS); }
    get loanAmount()  { return getFieldValue(this.loan.data, LOAN_AMOUNT); }
    get officerName() { return getFieldValue(this.loan.data, OFFICER_NAME); }

    get formattedAmount() {
        const amt = this.loanAmount;
        if(!amt) return '--';
        return new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD' }).format(amt);
    }

    get statusBadgeClass() {
        const map = { 'Approved': 'slds-badge slds-theme_success', 'Denied': 'slds-badge slds-theme_error' };
        return map[this.loanStatus] || 'slds-badge';
    }

    get isLoading() { return !this.loan?.data && !this.loan?.error; }

    // Refresh after updateRecord or DML:
    async handleAfterUpdate() {
        await refreshApex(this.wiredRecord); // re-fetches from server
    }
}
// IMPORTANT: getRecord only fires when recordId is truthy
// '$recordId' — the $ prefix means reactive — wire refires when recordId changes
// If recordId is undefined (component placed on non-record page), wire does not fire
How do you use updateRecord from LWC without a form — complete pattern?RECORD PAGE
// Inline update without lightning-record-edit-form
import { LightningElement, api, wire } from 'lwc';
import { updateRecord } from 'lightning/uiRecordApi';
import { ShowToastEvent } from 'lightning/platformShowToastEvent';
import { refreshApex } from '@salesforce/apex';
import LOAN_STATUS from '@salesforce/schema/Loan__c.Status__c';
import LOAN_ID from '@salesforce/schema/Loan__c.Id';

export default class LoanApproveButton extends LightningElement {
    @api recordId;

    async handleApprove() {
        // Build fields object — Id field is REQUIRED
        const fields = {};
        fields[LOAN_ID.fieldApiName] = this.recordId;
        fields[LOAN_STATUS.fieldApiName] = 'Approved';

        try {
            await updateRecord({ fields });
            this.dispatchEvent(new ShowToastEvent({
                title: 'Approved', message: 'Loan approved successfully', variant: 'success'
            }));
            // Force page to reload record data
            this.dispatchEvent(new CustomEvent('recordupdated'));
        } catch(error) {
            // Trigger addError() messages surface here automatically
            const msg = error.body?.output?.errors?.[0]?.message || error.body?.message;
            this.dispatchEvent(new ShowToastEvent({
                title: 'Approval Failed', message: msg, variant: 'error', mode: 'sticky'
            }));
        }
    }
}
How do you use createRecord and deleteRecord from LWC — with confirmation dialog?RECORD PAGE
import { createRecord, deleteRecord } from 'lightning/uiRecordApi';
import LOAN_OBJECT from '@salesforce/schema/Loan__c';
import LOAN_AMOUNT from '@salesforce/schema/Loan__c.LoanAmount__c';
import APPLICATION_FIELD from '@salesforce/schema/Loan__c.Application__c';

// CREATE RECORD
async handleCreateLoan() {
    const fields = {};
    fields[LOAN_AMOUNT.fieldApiName] = this.loanAmount;
    fields[APPLICATION_FIELD.fieldApiName] = this.recordId; // parent Id
    const recordInput = { apiName: LOAN_OBJECT.objectApiName, fields };

    try {
        const newRecord = await createRecord(recordInput);
        this.dispatchEvent(new ShowToastEvent({
            title: 'Loan Created', message: 'ID: ' + newRecord.id, variant: 'success'
        }));
        // Navigate to new record:
        this[NavigationMixin.Navigate]({
            type: 'standard__recordPage',
            attributes: { recordId: newRecord.id, actionName: 'view' }
        });
    } catch(e) {
        this.dispatchEvent(new ShowToastEvent({ title: 'Create Failed', message: e.body?.message, variant: 'error' }));
    }
}

// DELETE RECORD with confirmation
showDeleteConfirm = false;
handleDeleteClick() { this.showDeleteConfirm = true; }
handleCancelDelete() { this.showDeleteConfirm = false; }

async handleConfirmDelete() {
    this.showDeleteConfirm = false;
    try {
        await deleteRecord(this.recordId);
        this.dispatchEvent(new ShowToastEvent({ title: 'Deleted', variant: 'success' }));
        // Navigate back to list view after delete
        this[NavigationMixin.Navigate]({ type: 'standard__objectPage',
            attributes: { objectApiName: 'Loan__c', actionName: 'list' }
        });
    } catch(e) {
        this.dispatchEvent(new ShowToastEvent({ title: 'Delete Failed', message: e.body?.message, variant: 'error' }));
    }
}
How do you use the lwc:ref directive and this.refs — what replaced querySelector in modern LWC?RECORD PAGE

In API version 59.0+, Salesforce introduced lwc:ref as a cleaner alternative to this.template.querySelector() for accessing DOM elements.

<!-- OLD pattern — still works but less elegant -->
<lightning-input data-id="loanAmount" label="Loan Amount"></lightning-input>
// In JS: this.template.querySelector('[data-id="loanAmount"]')

<!-- NEW pattern — lwc:ref (API 59.0+) -->
<lightning-input lwc:ref="loanAmountInput" label="Loan Amount"></lightning-input>
<lightning-input lwc:ref="ficoInput" label="FICO Score"></lightning-input>

// In JS — use this.refs.refName
handleValidate() {
    const loanInput = this.refs.loanAmountInput;
    const ficoInput = this.refs.ficoInput;

    let isValid = loanInput.checkValidity() && ficoInput.checkValidity();
    if(!isValid) {
        loanInput.reportValidity();
        ficoInput.reportValidity();
    }
    return isValid;
}

// querySelectorAll still needed for multiple elements:
handleValidateAll() {
    const allInputs = this.template.querySelectorAll('lightning-input');
    return [...allInputs].every(input => {
        const valid = input.checkValidity();
        if(!valid) input.reportValidity();
        return valid;
    });
}

// Handle all 3 async scenarios properly
isLoading = true;  // start as true — wire fires immediately on load
error = undefined;
data = undefined;

// Scenario 1: Wire adapter
@wire(getLoanData, { loanId: '$recordId' })
wiredLoan({ data, error }) {
    this.isLoading = false;  // wire resolved — either data or error
    if(data) { this.data = data; this.error = undefined; }
    else if(error) { this.error = error.body?.message; }
}

// Scenario 2: Imperative call — fully manual isLoading control
async handleSearch() {
    this.isLoading = true;
    this.error = undefined;
    try {
        this.data = await searchLoans({ term: this.searchTerm });
    } catch(e) {
        this.error = e.body?.message || 'Search failed';
    } finally {
        this.isLoading = false; // ALWAYS in finally — even if error occurs
    }
}

// Scenario 3: Multiple parallel async calls
async loadAllData() {
    this.isLoading = true;
    try {
        const [loans, borrowers, docs] = await Promise.all([

        this.loans = loans; this.borrowers = borrowers; this.docs = docs;
    } catch(e) { this.error = e.body?.message; }
    finally { this.isLoading = false; }
}
