public class OpportunityTriggerHandler {
  public static void handle(List<Opportunity> newList, List<Opportunity> oldList,
                             System.TriggerOperation opType) {
    Set<Id> affectedAccountIds = new Set<Id>();

    if (newList != null) for (Opportunity o : newList) if (o.AccountId != null) affectedAccountIds.add(o.AccountId);
    if (oldList != null) for (Opportunity o : oldList) if (o.AccountId != null) affectedAccountIds.add(o.AccountId);

    if (affectedAccountIds.isEmpty()) return;

    OpportunityRollupService.updateOppCount(affectedAccountIds);

    // Send email only on new Opportunities
    if (opType == System.TriggerOperation.AFTER_INSERT) {
      OpportunityEmailService.notifyManagers(newList);
    }
  }
}

// ΓöÇΓöÇ STEP 4: Rollup Service ΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇ
public class OpportunityRollupService {
  public static void updateOppCount(Set<Id> accountIds) {
    // Aggregate count per account
    Map<Id, Integer> countMap = new Map<Id, Integer>();
    for (Id accId : accountIds) countMap.put(accId, 0); // default 0

    for (AggregateResult ar : [
        SELECT AccountId accId, COUNT(Id) cnt
        FROM Opportunity
        WHERE AccountId IN :accountIds AND StageName != 'Closed Lost'
        GROUP BY AccountId
    ]) {
      countMap.put((Id) ar.get('accId'), (Integer) ar.get('cnt'));
    }

    List<Account> toUpdate = new List<Account>();
    for (Id accId : accountIds) {
      toUpdate.add(new Account(Id=accId, Total_Opportunities__c=countMap.get(accId)));
    }
    update toUpdate;
  }
}

// ΓöÇΓöÇ STEP 5: Email Notification Service ΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇ
public class OpportunityEmailService {
  public static void notifyManagers(List<Opportunity> opps) {
    // Get Owner IDs
    Set<Id> ownerIds = new Set<Id>();
    for (Opportunity o : opps) ownerIds.add(o.OwnerId);

    // Get Owner + Manager details
    Map<Id, User> userMap = new Map<Id, User>(
      [SELECT Id, Name, Email, ManagerId, Manager.Name, Manager.Email
       FROM User WHERE Id IN :ownerIds AND ManagerId != null]
    );

    // Build emails — one per Opportunity
    List<Messaging.SingleEmailMessage> emails = new List<Messaging.SingleEmailMessage>();
    for (Opportunity o : opps) {
      User owner = userMap.get(o.OwnerId);
      if (owner == null || owner.ManagerId == null) continue; // skip if no manager

      Messaging.SingleEmailMessage mail = new Messaging.SingleEmailMessage();
      mail.setToAddresses(new String[]{ owner.Manager.Email });
      mail.setSubject('New Opportunity Created: ' + o.Name);
      mail.setPlainTextBody(
        'Hello ' + owner.Manager.Name + ',\n\n' +
        'A new Opportunity has been created by your team member ' + owner.Name + ':\n\n' +
        'Opportunity Name: ' + o.Name + '\n' +
        'Amount: $' + (o.Amount != null ? String.valueOf(o.Amount) : 'Not set') + '\n' +
        'Stage: ' + o.StageName + '\n' +
        'Close Date: ' + String.valueOf(o.CloseDate) + '\n\n' +
        'Please review in Salesforce.\n\n' +
        'Regards,\n' +
        'Salesforce Automation'
      );
      emails.add(mail);
    }

    if (!emails.isEmpty()) Messaging.sendEmail(emails);
  }
}
Why this design

Separating rollup logic (OpportunityRollupService) from email logic (OpportunityEmailService) follows the Single Responsibility principle — each can be tested and changed independently.

ΓÅ░ Cron Expressions in Salesforce
Cron expressions in Salesforce Scheduled Apex — complete referenceSCHEDULED APEX

Salesforce cron expression has 7 positions (not 5 like standard Unix cron):

// Position:  Seconds  Minutes  Hours  Day-of-Month  Month  Day-of-Week  Year
// Format:    0-59     0-59     0-23   1-31          1-12   1-7 or SUN-SAT  (optional)

// Special characters:
// *  = any value
// ?  = no specific value (use for Day-of-Month OR Day-of-Week — not both)
// -  = range (e.g., MON-FRI)
// ,  = list (e.g., MON,WED,FRI)
// /  = increment (e.g., 0/15 = every 15 starting from 0)
// L  = last (e.g., L in Day-of-Month = last day of month)

// COMMON EXAMPLES:
'0 0 2 * * ?'           // Every day at 2:00 AM
'0 0 2 ? * MON'         // Every Monday at 2:00 AM
'0 0 2 1 * ?'           // First day of every month at 2:00 AM
'0 0 0 1 1 ? *'         // January 1st every year at midnight
'0 0/30 * * * ?'        // Every 30 minutes (avoid — use PE instead)
'0 0 8-17 ? * MON-FRI'  // Every hour 8AM-5PM Mon-Fri
'0 0 2 L * ?'           // Last day of every month at 2AM

// SCHEDULE AN APEX JOB:
String cronExpr = '0 0 2 * * ?'; // 2AM daily
String jobName  = 'Nightly Loan Sync';
System.schedule(jobName, cronExpr, new NightlyLoanSyncSchedule());

// CHECK SCHEDULED JOBS:
List<CronTrigger> jobs = [
  SELECT Id, CronJobDetail.Name, CronExpression, State, NextFireTime
  FROM CronTrigger WHERE CronJobDetail.Name = :jobName
];

// ABORT A SCHEDULED JOB:
for (CronTrigger ct : jobs) System.abortJob(ct.Id);

// QUERY ALL ACTIVE SCHEDULED JOBS:
[SELECT Id, CronJobDetail.Name, NextFireTime, State FROM CronTrigger
 WHERE State NOT IN ('DELETED','COMPLETE') ORDER BY NextFireTime]
Frequency limit

Do not schedule jobs more frequently than once per hour using Scheduled Apex. For higher frequency automation, use Platform Events + trigger subscribers. Salesforce limits orgs to 100 scheduled jobs total.

public class LoanStatusAction {

  public class Request {
    @InvocableVariable(label='Loan Number' required=true)
    public String loanNumber;
  }

  public class Response {
    @InvocableVariable(label='Status')      public String status;
    @InvocableVariable(label='Amount')      public Decimal amount;
    @InvocableVariable(label='Next Steps')  public String nextSteps;
    @InvocableVariable(label='Found')       public Boolean found;
  }

  @InvocableMethod(
    label='Get Loan Status'
    description='Returns the current status, loan amount, and recommended next steps for a given loan number. Use when the borrower asks about their loan status or progress.'
  )
  public static List<Response> getLoanStatus(List<Request> requests) {
    List<Response> responses = new List<Response>();

    for (Request req : requests) {
      Response res = new Response();
      List<Loan__c> loans = [SELECT Name, Status__c, Amount__c
          FROM Loan__c WHERE Name = :req.loanNumber WITH SECURITY_ENFORCED LIMIT 1];

      if (loans.isEmpty()) {
        res.found = false; res.nextSteps = 'Loan number not found. Please verify and try again.';
      } else {
        res.found     = true;
        res.status    = loans[0].Status__c;
        res.amount    = loans[0].Amount__c;
        res.nextSteps = getNextSteps(loans[0].Status__c);
      }
      responses.add(res);
    }
    return responses;
  }

  private static String getNextSteps(String status) {
    Map<String,String> steps = new Map<String,String>{
      'Applied'       => 'Application received. Credit check within 24 hours.',
      'Credit_Approved' => 'Credit approved. Underwriter review in progress.',
      'Approved'      => 'Loan approved! Closing being scheduled.',
      'Funded'        => 'Congratulations — your loan is funded!'
    };
    return steps.containsKey(status) ? steps.get(status) : 'Contact your loan officer.';
  }
}

LWC Performance optimization — what are the common issues and fixes?LWC PERFORMANCE
  • Too many @wire calls on load — load only what's needed immediately. Lazy-load secondary data on user action (button click, tab change) using imperative Apex.
  • Re-rendering on every keystroke — debounce input handlers. Don't fire Apex on every keypress, wait 300ms after last keystroke.
  • Large datatable (1000+ rows) — use infinite scroll with pagination instead of loading all rows at once. lightning-datatable with enable-infinite-loading.
  • Unnecessary tracked objects — use @track only when you mutate nested properties. Primitives re-render without @track.
  • Missing key in for:each — always set key attribute to a unique stable ID. Missing/wrong key causes full list re-render on any change.
