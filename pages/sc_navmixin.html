How do you import and use NavigationMixin in LWC — what is the basic pattern?NAVMIXIN
import { LightningElement, api } from 'lwc';
import { NavigationMixin } from 'lightning/navigation';

// ALWAYS extend NavigationMixin(LightningElement)
export default class MyNavComp extends NavigationMixin(LightningElement) {
    @api recordId;

    // Navigate to the current record's view page
    handleViewRecord() {
        this[NavigationMixin.Navigate]({
            type: 'standard__recordPage',
            attributes: {
                recordId: this.recordId,
                actionName: 'view'   // 'view', 'edit', 'clone'
            }
        });
    }

    // Generate a URL without navigating (for links, sharing)
    async handleGetUrl() {
        const url = await this[NavigationMixin.GenerateUrl]({
            type: 'standard__recordPage',
            attributes: { recordId: this.recordId, actionName: 'view' }
        });
        // url = '/lightning/r/Loan__c/001xxx/view'
        const fullUrl = window.location.origin + url;
        navigator.clipboard.writeText(fullUrl);
    }
}

// RECEIVER — read state from CurrentPageReference
import { CurrentPageReference } from 'lightning/navigation';

@wire(CurrentPageReference)
pageRef(ref) {
    if (ref) {
        this.loanId = ref.state?.loanId;
        this.mode = ref.state?.mode;
        this.sourceApp = ref.state?.sourceApp;
    }
}
State vs Attributes

state becomes query parameters in the URL (visible, bookmarkable). attributes is part of the route structure. Use state for optional context data, attributes for required routing parameters like recordId.

How do you navigate to a related record from a parent record page — get the related record ID first?NAVMIXIN
import { LightningElement, api, wire } from 'lwc';
import { NavigationMixin } from 'lightning/navigation';
import { getRecord, getFieldValue } from 'lightning/uiRecordApi';
import APPLICATION_FIELD from '@salesforce/schema/Loan__c.Application__c';
import APPLICATION_NAME from '@salesforce/schema/Loan__c.Application__r.Name';

export default class LoanParentNav extends NavigationMixin(LightningElement) {
    @api recordId;
    applicationId;

    @wire(getRecord, { recordId: '$recordId', fields: [APPLICATION_FIELD, APPLICATION_NAME] })
    wiredLoan({ data }) {
        if (data) {
            this.applicationId = getFieldValue(data, APPLICATION_FIELD);
        }
    }

    handleViewApplication() {
        if (!this.applicationId) return;
        this[NavigationMixin.Navigate]({
            type: 'standard__recordPage',
            attributes: {
                recordId: this.applicationId,
                objectApiName: 'Application__c',
                actionName: 'view'
            }
        });
    }
}

How do you create a shareable link to a record and copy it to clipboard?NAVMIXIN
import { LightningElement, api } from 'lwc';
import { NavigationMixin } from 'lightning/navigation';
import { ShowToastEvent } from 'lightning/platformShowToastEvent';

export default class ShareLoanLink extends NavigationMixin(LightningElement) {
    @api recordId;
    shareableUrl = '';

    async connectedCallback() {
        try {
            const relativeUrl = await this[NavigationMixin.GenerateUrl]({
                type: 'standard__recordPage',
                attributes: { recordId: this.recordId, actionName: 'view' }
            });
            // GenerateUrl returns relative path — prepend origin for absolute URL
            this.shareableUrl = window.location.origin + relativeUrl;
        } catch(e) {
            console.error('URL generation failed:', e);
        }
    }

    handleCopyLink() {
        navigator.clipboard.writeText(this.shareableUrl).then(() => {
            this.dispatchEvent(new ShowToastEvent({
                title: 'Link Copied',
                message: 'Loan page link copied to clipboard',
                variant: 'success'
            }));
        });
    }
}
What happens when NavigationMixin.Navigate is called on a component not in Lightning Experience — how do you handle it?NAVMIXIN

NavigationMixin works in Lightning Experience, Salesforce Mobile App, and Experience Cloud sites. It does NOT work in Visualforce pages, standalone apps outside the Salesforce container, or in Jest tests without mocking.

// Guard navigation with a try-catch
handleNavigate() {
    try {
        this[NavigationMixin.Navigate]({
            type: 'standard__recordPage',
            attributes: { recordId: this.recordId, actionName: 'view' }
        });
    } catch(e) {
        // Fallback for non-Lightning contexts
        window.location.href = '/lightning/r/Loan__c/' + this.recordId + '/view';
    }
}

// In Jest tests — mock NavigationMixin
// jest.config.js moduleNameMapper:
// '@salesforce/schema': '/__mocks__/sfSchema.js'
// 'lightning/navigation': '/__mocks__/navigation.js'

// __mocks__/navigation.js:
const NavigationMixin = (Base) => {
    class NavigationMixinClass extends Base {
        navigate = jest.fn();
        [Symbol.hasInstance](instance) { return instance instanceof NavigationMixinClass; }
    }
    NavigationMixinClass.Navigate = Symbol('Navigate');
    NavigationMixinClass.GenerateUrl = Symbol('GenerateUrl');
    return NavigationMixinClass;
};
export { NavigationMixin };
How do you track page navigation in LWC — react when the user navigates to a different record?NAVMIXIN
// CurrentPageReference wire fires every time the page URL changes
import { CurrentPageReference } from 'lightning/navigation';

@wire(CurrentPageReference)
pageRef(newRef) {
    if (!newRef) return;
    const newRecordId = newRef.attributes?.recordId;
    if (newRecordId && newRecordId !== this.currentRecordId) {
        this.currentRecordId = newRecordId;
        this.handleRecordChange(newRecordId);
    }
}

handleRecordChange(recordId) {
    // Reload data for the new record
    this.isLoading = true;
    getLoanData({ loanId: recordId })
        .then(data => { this.loanData = data; })
        .catch(err => { this.error = err.body?.message; })
        .finally(() => { this.isLoading = false; });
}

How do you navigate programmatically after a successful record save in LWC?NAVMIXIN
import { LightningElement, api } from 'lwc';
import { NavigationMixin } from 'lightning/navigation';
import { createRecord } from 'lightning/uiRecordApi';
import { ShowToastEvent } from 'lightning/platformShowToastEvent';
import LOAN_OBJECT from '@salesforce/schema/Loan__c';
import LOAN_AMOUNT from '@salesforce/schema/Loan__c.LoanAmount__c';

export default class LoanCreator extends NavigationMixin(LightningElement) {
    @api applicationId;

    async handleSave() {
        try {
            const fields = {
                [LOAN_AMOUNT.fieldApiName]: this.loanAmount,
                'Application__c': this.applicationId
            };
            const newRecord = await createRecord({ apiName: LOAN_OBJECT.objectApiName, fields });

            // Show success toast
            this.dispatchEvent(new ShowToastEvent({
                title: 'Loan Created', variant: 'success',
                message: 'Loan ' + newRecord.id + ' created successfully'
            }));

            // Navigate to the new record immediately after creation
            this[NavigationMixin.Navigate]({
                type: 'standard__recordPage',
                attributes: {
                    recordId: newRecord.id,
                    objectApiName: 'Loan__c',
                    actionName: 'view'
                }
            });
        } catch(e) {
            this.dispatchEvent(new ShowToastEvent({
                title: 'Save Failed', message: e.body?.message, variant: 'error'
            }));
        }
    }
}
